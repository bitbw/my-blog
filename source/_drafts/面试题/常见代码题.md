## 防抖节流

```js
/**
 * @description:  防抖
 * @param {*} fn
 * @param {*} delay
 * @param {*} key 是否立即执行
 * @return {*}
 */
function debounce(fn, delay, key = false) {
  let timer;
  return function (...arg) {
    const context = this;
    if (timer) clearTimeout(timer);
    if (key) {
      timer = setTimeout(() => {
        timer = null;
      }, delay);
      if (!timer) {
        fn.call(context, ...arg);
      }
    } else {
      timer = setTimeout(() => {
        fn.call(context, ...arg);
      }, delay);
    }
  };
}
/**
 * @description:  节流
 * @param {*} fn
 * @param {*} delay
 * @param {*} key 是否定时器版
 * @return {*}
 */
function throttle(fn, delay, key) {
  let perTimer = 0;
  let timer;
  return function (...arg) {
    const context = this;
    if (key) {
      if (!timer) {
        timer = setTimeout(() => {
          fn.call(context, ...arg);
          timer = null;
        }, delay);
      }
    } else {
      const curTimer = Date.now();
      if (curTimer - perTimer > delay) {
        fn.call(context, ...arg);
        perTimer = curTimer;
      }
    }
  };
}
```

## 一维数组构建树状结构

```js
var nodes = [
  { id: 2, name: "节点2", pid: 6 },
  { id: 3, name: "节点3", pid: 1 },
  { id: 6, name: "节点6", pid: null },
  { id: 4, name: "节点4", pid: 3 },
  { id: 1, name: "节点1", pid: 2 },
  { id: 7, name: "节点7", pid: 2 },
  { id: 10, name: "节点10", pid: 7 },
  { id: 9, name: "节点9", pid: 8 },
  { id: 8, name: "节点8", pid: 1 },
];

function buildTree(list = []) {
  function getChild(nodeList, arrList) {
    for (const node of nodeList) {
      node.children = arrList.filter((item) => item.pid === node.id);
      if (node.children.length) {
        getChild(node.children, arrList);
      }
    }
  }
  const treeList = list.filter((item) => !item.pid);
  getChild(treeList, list);
  return treeList;
}
buildTree(nodes);

    }
  }
  return treeList;
}
```
